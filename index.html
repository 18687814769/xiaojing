创建图片 Clip clip = ImageClip(img_path).set_duration(duration) # 添加 Ken Burns 效果 (缩放) # 简单实现：使用 resize 和 crop 模拟推拉 # 这里为了简化，先做静态展示，后续可优化为动态缩放 # clip = clip.resized(height=1080) # 假设需要调整尺寸 # 添加字幕 txt = TextClip(narrations[i], fontsize=24, color='white', font='Arial', stroke_color='black', stroke_width=1, size=(clip.size[0]*0.9, None), method='caption') txt = txt.set_pos(('center', 'bottom')).set_duration(duration) clip = CompositeVideoClip([clip, txt]) clips.append(clip) #
